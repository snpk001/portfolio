sql1 = 'SELECT date_part(\'year\', age(b.term_start, a.first_term)) as period, count (distinct a.id_bioguide) \
FROM \
(\
    SELECT id_bioguide, min(term_start) as first_term \
    FROM \"congress\".legislators_terms \
    GROUP BY 1 \
    ) a \
JOIN \"congress\".legislators_terms b on a.id_bioguide = b.id_bioguide \
GROUP BY 1 order by 1 \
;'
